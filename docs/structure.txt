NearTheLake/
│
├── README.md
├── assets/                        # Статичні файли (зображення)

│
├── bot/                           # Основна логіка Telegram-бота
│   ├── command_layer/             # Обробка текстових команд користувача
│   │   ├── command_managers/      # Менеджери команд за станами гри
│   │   ├── commands/              # Команди: старт, правила, налаштування тощо
│   │   ├── dispatcher.py          # Делегує команду у відповідний менеджер
│   │   ├── router.py              # Визначає менеджер для стану гравця
│   ├── handlers/                  # Telegram-хендлери: команди, callback-и, повідомлення
│   ├── main_router.py             # Реєстрація хендлерів
│   ├── services/                  # Сервіси для бізнес-логіки: сесії, месенджер, наради
│   ├── ui_components/            
│       ├── keyboard.py           # Генерація клавіатур (Reply)
│       ├── promt_action.py       # Відображення кнопок для вибору дій
│
├── config/                        # Конфігурація та глобальні константи
│   ├── constants.py               # Сталі: назви дій, параметри
│   ├── helpers.py                 # Доступ до глобальних об'єктів (гра тощо)
│   ├── secrets.py                 # TOKEN бота
│   ├── settings.py                # Глобальні значення за замовчуванням (можна винести до GameSettings)
│
├── game/                          # Логіка гри
│   ├── actions/                   # Реалізація кожної дії: скидання, очищення, бонус, штраф
│   ├── core/                      # Основні об'єкти гри: Game, GameSettings, Lake, Player
│   ├── events/                    # Спецподії: наради, паводок
│   ├── gamelogic/                 # Обробка ходу гри: застосування дій, генерація результатів
│
├── messages/                      # Локалізовані повідомлення для UI (винесено окремо)
│
├── main.py                        # Точка входу: створення Application, запуск
